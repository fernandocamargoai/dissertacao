\begin{frame}
  \frametitle{Equações de geração de energia}
  
  Geração Hidráulica:
  \begin{equation}\label{eq:geracaoHidraulica}
    gh = k.(h_{mon}(v) - h_{jus}(u) - pc).q
  \end{equation}
  \begin{equation}\label{eq:constanteK}
    k=\bar{\eta}.pa.g
  \end{equation}
  Onde:
  
  \begin{description}
    \item[$gh$] geração hidráulica [$W$]
    \item[$h_{mon}(v)$] altura a montante [$m$]
    \item[$h_{jus}(u)$] altura a jusante [$m$]
    \item[$pc$] perdas hidráulicas
    \item[$k$] produtibilidade da usina
    \item[$\bar{\eta}$] rendimento médio da usina
    \item[$pa$] peso específico da água [$N/m^{3}$]
    \item[$g$] gravidade [$m/s^{2}$]
  \end{description}
\end{frame}
\end{comment}

\begin{frame}
  \frametitle{Formulação do Problema}
  
  Funções Objetivo:
  \begin{equation}\label{eq:funcoesObjetivo}
    \vec{F} = \{CT, VF\}
  \end{equation}
  
  \begin{equation}\label{eq:funcaoObjetivoCustoTermico}
    CT = min \sum_{t=1}^{T} f(gt_{t})
  \end{equation}
  \begin{equation}\label{eq:funcaoObjetivoCustoTermico}
    VF = max \sum_{i=1}^{N} v_{iT}
  \end{equation}
\end{frame}

\begin{comment}
\begin{frame}
  \frametitle{Formulação do Problema}
  
  Onde:
  
  \begin{description}
    \item[$\vec{F}$] função multiobjetivo
    \item[$CT$] objetivo de minimizar custo térmico total
    \item[$VF$] objetivo de maximizar volume final dos reservatórios
    \item[$T$] número de estágios do horizonte de planejamento (intervalo de tempo) [meses]
    \item[$N$] número de usinas hidrelétricas do sistema
    \item[$t$] unidade de tempo
    \item[$i$] índice da usina hidrelétrica do sistema
    \item[$gt_{t}$] geração térmica em $t$
    \item[$f(gt_{t})$] custo térmico em $t$
    \item[$v_{iT}$] volume final da hidrelétrica $i$
  \end{description}
\end{frame}
\end{comment}

\begin{frame}
  \frametitle{Restrições}
  
  Balanceamento de carga:
  \begin{equation}\label{eq:balanceamentoDeCarga}
    gt_{t} + \sum_{i=1}^{N} gh_{it} - D_{t} - L_{t} = 0
  \end{equation}
  
  Balanceamento dinâmico de água:
  \begin{equation}\label{eq:balanceamentoDinamicoDaAgua}
    v_{it} = v_{i,t-1} + y_{it} - q_{it} - spl_{it} + \sum_{m=1}^{\Phi_{i}} (q_{m,t} + spl_{m,t}), m \in \Phi_{i}
  \end{equation}
  
  Demais restrições: limites de geração térmica, limites de geração hidrelétrica, limites de armazenamento e limites de turbinagem.
\end{frame}  

\begin{comment}
\begin{frame}
  \frametitle{Restrições}
  
  Onde:
  
  \begin{description}
    \item[$gh_{it}$] geração da usina $i$ durante o período $t$ [$\overline{\rm MW}$]
    \item[$D_{t}$] demanda a ser atendida durante o período $t$ [$\overline{\rm MW}$]
    \item[$L_{t}$] perdas na transmissão de energia no instante $t$ [$\overline{\rm MW}$]
    \item[$v_{it}$] volume do reservatório da usina $i$ no final do intervalo $t$ [$hm^{3}$]
    \item[$y_{it}$] vazão incremental afluente à usina $i$ durante o intervalo $t$ [$m^{3}/s$]
    \item[$q_{it}$] vazão turbinada pela usina $i$ durante o intervalo $t$ [$m^{3}$]
    \item[$spl_{it}$] vazão vertida pela usina $i$ durante o intervalo $t$ [$m^{3}$]
    \item[$\Phi_{i}$] conjunto de usinas imediatamente a montante da usina $i$
  \end{description}
\end{frame}
\end{comment}

\subsection{Métricas de Qualidade}

\begin{frame}
  \frametitle{Métricas de Qualidade}
    
  \begin{outline}
  \0\onslide<1->{Tipos de Indicador de Qualidade:}
    \1<1-> Unário: visa quantificar a qualidade de uma aproximação em relação a uma Fronteira de Pareto de referência
    \1<2-> Binário: visa comparar duas aproximações diferentes.
  \0\onslide<3->{Aspectos quantificados:}
    \1<3-> Convergência
    \1<4-> Diversidade
  \0\onslide<5->{Métricas utilizadas:}
    \1<5-> Epsilon $\epsilon+$: avalia convergência
    \1<6-> SPREAD ($\Delta$): avalia diversidade
    \1<7-> Hypervolume (HV): avalia convergência e diversidade
  \end{outline}
\end{frame}

\begin{comment}
\begin{frame}
  \frametitle{Epsilon Aditivo Unário \texorpdfstring{($\epsilon+$)}{}}
    
  \begin{outline}
    \1<1-> Tipo: unário
    \1<2-> Aspecto avaliado: convergência
    \1<3-> Valor a ser minimizado ($0$ indica que $A$ estritamente domina $R$)
    \1<4-> Trabalha ponto a ponto
    \1<5-> Calcula-se a parcela mínima $\epsilon$ a qual pode ser somada a qualquer ponto de $R$ de forma que $A$ passe a dominá-lo fracamente
  \end{outline}
\end{frame}

\begin{frame}
  \frametitle{SPREAD \texorpdfstring{($\Delta$)}{(Delta)}}
    
  \begin{outline}
    \1<1-> Tipo: unário
    \1<2-> Aspecto avaliado: diversidade
    \1<3-> Valor a ser minimizado ($0$ indica que $A$ é abrangente e uniformemente espalhado)
    \1<4-> Trabalha ponto a ponto
    \1<5-> Calcula-se distâncias euclidianas ponto a ponto, a média dessas distâncias e então aplica-se um cálculo em cima disso
  \end{outline}
\end{frame}

\begin{frame}
  \frametitle{Hipervolume (HV)}
    
  \begin{outline}
    \1<1-> Tipo: unário
    \1<2-> Aspecto avaliado: convergência e diversidade
    \1<3-> Valor a ser maximizado
    \1<4-> Trabalha com ponto de referência
    \1<5-> Mede o tamanho do espaço objetivo coberto por uma aproximação A
  \end{outline}
\end{frame}
\end{comment}

    \1<2-> Seleção de pais: torneio binário entre membros do arquivo externo
    \1<3-> Nova geração: filhos
    \1<4-> Considerado em estado de arte

      \2 Arquivo também usado como líderes do enxame

    \1<3-> Adiciona mutação às partículas
      \2 Mutação uniforme
      \2 Mutução não uniforme
      \2 Nenhuma mutação
    \1<4-> $\epsilon$-dominância
    \1<5-> Considerado o MOPSO mais proeminente antes do SMPSO
