\begin{frame}
  \frametitle{Sistema Hidrotérmico}
  
  \begin{outline}
    \1<1-> Hidrelétricas
      \2 Fonte primária de energia
      \2 Custo fixo de manutenção, independente da energia produzida
    \1<2-> Termelétricas
      \2 Fonte complementar
      \2 Custo variável de acordo com preço de combustíveis e energia produzida
  \end{outline}
\end{frame}

\begin{frame}
  \frametitle{Crises energéticas brasileiras}
  
  \begin{outline}
    \1<1-> 2001 (Apagão)
      \2 Forte período de seca
      \2 Reservatórios em níveis muito baixos
      \2 Racionamento de energia nacional
      \2 Causas:
	\3 Esgotamento do modelo estatal
	\3 Falhas no planejamento da transição do modelo estatal para o modelo privado
	\3 Problemas contratuais e regulatórios
	\3 Falta de coordenação entre os órgãos governamentais
    \1<2-> 2013
      \2 Período de seca menor que da crise anterior
      \2 Reservatórios com níveis iguais ou abaixo da crise anterior
      \2 Ameaça de racionamento de energia nacional
      \2 Despacho maciço de usinas térmicas
      \2 Grande impacto econômico
      \2 Causas: desconhecidas
  \end{outline}
\end{frame}

\begin{frame}
  \frametitle{O Problema do Planejamento Energético de Sistemas Hidrotérmicos}
  
  Considerações:
  \begin{outline}
    \1<1-> Decisões de operação afetam decisões futuras: reservatórios
    \1<2-> Acoplamento de usinas na mesma bacia
    \1<3-> Aleatoriedade das vazões
    \1<4-> Objetivos:
      \2<4-> Minimizar custo de produção (custo térmico)
      \2<5-> Definir uma estratégia de geração para cada usina de interconectada
      \2<6-> Aumentar produção hidrelétrica para diminuir termelétrica
      \2<7-> Atender demanda com confiabilidade
  
  \end{outline}
\end{frame}
